    public void RB_INSERT_FIXUP(RedBlackTree T, RedBlackTreeNode z){
        //ÂΩìzÁöÑÁà∂ÁªìÁÇπ‰∏∫Á∫¢Ëâ≤Êó∂ÔºåÊèíÂÖ•ÁªìÁÇπÂíåÁà∂ÁªìÁÇπÂêå‰∏∫Á∫¢Ëâ≤ÔºåÈúÄË¶ÅË∞ÉÊï¥
        while( z.getParent().getColor() == NodeColor.Red ){

            if( z.getParent() == z.getParent().getParent().getLeft() ){
                //yÂÆö‰πâ‰∏∫ÂèîÂèîÁªìÁÇπ
                RedBlackTreeNode y = z.getParent().getParent().getRight();
                //case1ÔºöÂ¶ÇÊûúÂèîÂèîÁªìÁÇπ‰∏∫Á∫¢Ëâ≤Ôºå‰∏äÂ±ÇÁà∂ÁªìÁÇπÂíåÂèîÂèîÁªìÁÇπËÆæ‰∏∫ÈªëËâ≤ÔºåÁ•ñÁà∂ËäÇÁÇπËÆæ‰∏∫Á∫¢Ëâ≤
                if( y.getColor() == NodeColor.Red ){
                    z.getParent().setColor(NodeColor.Black);
                    y.setColor(NodeColor.Black);
                    z.getParent().getParent().setColor(NodeColor.Red);
                    //Á•ñÁà∂ÁªìÁÇπËÆæ‰∏∫z
                    z = z.getParent().getParent();
                }
                //case2ÔºöÊèíÂÖ•ÁªìÁÇπ‰∏∫Áà∂ÁªìÁÇπÁöÑÂè≥Â≠êÊ†ëÔºåÔºàÂèîÂèîÁªìÁÇπ‰∏ÄÂÆö‰∏∫ÈªëËâ≤Ôºâ,Áà∂ÁªìÁÇπËÆæ‰∏∫zÔºåÂØπzÂ∑¶Êóã
                else if( z == z.getParent().getRight() ){
                    //ÂØπÁà∂ÁªìÁÇπÂ∑¶Êóã
                    z = z.getParent();
                    LEFT_ROTATE(T, z);
                }
                //case3ÔºöÊèíÂÖ•ÁªìÁÇπ‰∏∫Áà∂ÁªìÁÇπÁöÑÂ∑¶Â≠êÊ†ëÔºåÔºàÂèîÂèîÁªìÁÇπ‰∏ÄÂÆö‰∏∫ÈªëËâ≤ÔºâÔºåÁà∂ÁªìÁÇπËÆæ‰∏∫ÈªëËâ≤ÔºåÁ•ñÁà∂ÁªìÁÇπËÆæ‰∏∫Á∫¢Ëâ≤ÔºåÂØπÁ•ñÁà∂ÁªìÁÇπÂè≥Êóã
                z.getParent().setColor(NodeColor.Black);
                z.getParent().getParent().setColor(NodeColor.Red);
                RIGHT_ROTATE(T, z.getParent().getParent());
            }
            //ÊèíÂÖ•ÁªìÁÇπÁöÑÁà∂ÁªìÁÇπÊòØÁ•ñÁà∂ËäÇÁÇπÁöÑÂè≥Â≠êÊ†ëÔºåÂêåÁêÜÔºåÊñπÂêë‰∫§Êç¢
            else{
                RedBlackTreeNode y = z.getParent().getParent().getLeft();
                if( y.getColor() == NodeColor.Red ){
                    z.getParent().setColor(NodeColor.Black);
                    y.setColor(NodeColor.Black);
                    z.getParent().getParent().setColor(NodeColor.Red);
                    z = z.getParent().getParent();
                }
                else if( z == z.getParent().getLeft() ){
                    z = z.getParent();
                    RIGHT_ROTATE(T, z);
                }
                z.getParent().setColor(NodeColor.Black);
                z.getParent().getParent().setColor(NodeColor.Red);
                LEFT_ROTATE(T, z.getParent().getParent());
            }
        }
        T.getRoot().setColor(NodeColor.Black);
    }

    public void RB_DELETE(RedBlackTree T, RedBlackTreeNode z){

        if( IsNil(z.getLeft()) == true ){

        //case3ÔºözÊúâÂ∑¶Âè≥Â≠êÊ†ë
        else{
            //ÊâæÂà∞Âè≥Â≠êÊ†ë‰∏≠ÊúÄÂ∞èÁöÑÁªìÁÇπÔºåÂç≥zÁöÑÂêéÁªß
            y = TREE_MINIMUM(z.getRight());

            x = y.getRight();

            if( y.getParent() == z ){
                x.setParent(y);
            }
            //Ëã•‰∏çÁõ∏Ëøû
            else{
                RB_TRANSPLANT(T, y, y.getRight());
                y.setRight(z.getRight());
                y.getRight().setParent(y);
            }
            RB_TRANSPLANT(T, z, y);
            y.setLeft(z.getLeft());
            y.getLeft().setParent(y);
            y.setColor(z.getColor());
        }
        //Âà†Èô§ÁªìÁÇπ‰∏∫ÈªëËâ≤Êó∂ÈúÄË¶ÅË∞ÉÊï¥
        if( yOriginColor == NodeColor.Black ){
            RB_DELETE_FIXUP(T, x);
        }
    }

    public void RB_DELETE_FIXUP(RedBlackTree T, RedBlackTreeNode x){
