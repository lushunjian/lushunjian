func mergeNums(nums1 []int,nums2 []int,m int,n int){
   for i:=0;i<n;i++{
      e := nums2[i]
      var index = 0
      for j := m-1+i;j>0 && nums1[j]>e;j--{
         nums1[j+1] = nums1[j]
         index = j
      }
      nums1[index] = e
   }
}

func water(array []int) int {
   i,j :=0,len(array)-1
   w,h := 0,0
   var max = 0
   for ;i<j; {
      w = j-i
      temp1 := array[i]
      temp2 := array[j]
      if temp1>temp2{
         h = temp2
      } else {
         h = temp1
      }
      area := w*h
      if area>max {
         max = area
      }

      if temp1>temp2{
         j--
      }else {
         i++
      }
   }
   return max
}

   var max = 0
   for i:=0;i<len(array);i++{
